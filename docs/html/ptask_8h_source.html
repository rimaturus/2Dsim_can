239 /*
240  * Returns the elapsed time (since ptask_init) in the specified unit.
241  * Use MICRO for microseconds and NANO for nanoseconds.
242  */
243 long get_systime(int unit)
244 {
245  timespec_custom t;
246  long tu, mul, div;
247  switch (unit) {
248  case MICRO: mul = 1000000; div = 1000; break;
249  case NANO: mul = 1000; div = 1000000; break;
250  default: mul = 1000; div = 1000000; break;
251  }
252  clock_gettime(CLOCK_MONOTONIC, (struct timespec *)&t);
253  tu = (t.tv_sec - ptask_t0.tv_sec) * mul;
254  tu += (t.tv_nsec - ptask_t0.tv_nsec) / div;
255  return tu;
256 }
